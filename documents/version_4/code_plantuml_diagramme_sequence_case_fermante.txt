@startuml

activate Jeu
Jeu -> Labyrinthe : getCase()
activate Labyrinthe
Labyrinthe --> Jeu : case : Case
deactivate Labyrinthe

Jeu -> Case : getType()
activate Case
Case --> Jeu : type : int
deactivate Case

alt si c'est une case a effet
  Jeu -> CaseAEffet : activerEffet(Entite e)
  activate CaseAEffet
  CaseAEffet -> CaseSecrete : activerEffet(Entite e)
  activate CaseSecrete
  CaseSecrete -> Porte : changerEtat(false)
  activate Porte
  Porte --> CaseSecrete
  deactivate Porte
  CaseSecrete --> CaseAEffet
  deactivate CaseSecrete
  CaseAEffet --> Jeu
  deactivate CaseAEffet
end

@enduml
